CREATE VIEW flightsfromto AS 
SELECT f.volID, f.departAirport, t.destAirport
FROM flightsfrom f
INNER JOIN 
flightsto t
ON f.volID= t.volID

//flights tables
CREATE VIEW flights AS 
SELECT v.volID, v.departureDate, t.departAirport, t.destAirport, v.arrivalDate, v.price
FROM vols v
INNER JOIN 
flightsfromto t
ON v.volID= t.volID

//CREATE reservations table


CREATE VIEW reservations AS
SELECT r.reservID, p.fName, p.lName, f.departureDate, f.departAirport, f.destAirport, f.arrivalDate, r.dateReserv, r.seatNum, f.price, r.goingComing
FROM reservs r
INNER JOIN 
passengers p
ON r.passengerID=p.passengerID
INNER JOIN 
flights f
ON f.volID=p.volID
